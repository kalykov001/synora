👤 profiles
id (uuid, FK → auth.users)
full_name text
avatar_url text
language text (default 'en')
created_at timestamptz
📄 documents
id uuid PK
user_id uuid FK
title text
file_path text (Storage)
language_detected text
page_count int
status text (processing/ready)
created_at timestamptz
📝 summaries
id uuid PK
document_id uuid FK
content text
language text
created_at timestamptz
🃏 flashcard_sets + items
flashcard_sets:
id, document_id FK, title

flashcard_items:
id, set_id FK, question,

exam_questions: id, exam_id,

exam_attempts: id, exam_id,
 user_id, score, answers[]

tutor_sessions:
id, user_id, document_id (opt)

tutor_messages:
id, session_id FK, role
(user/assistant), content
